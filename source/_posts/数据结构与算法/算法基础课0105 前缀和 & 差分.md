# 前缀和 + 差分

## 前缀和

一维前缀和

```c++
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 1e5+10;
int a[N];

int main(){
  int n,m;
  cin>>n>>m;
  for(int i=1;i<=n;i++){
    int x;
    scanf("%d",&x);
    a[i]=a[i-1]+x;
  }
  while(m--){
    int l,r;
    scanf("%d%d",&l,&r);
    printf("%d\n",a[r]-a[l-1]);
  }
  return 0;
}
```

* 二维前缀和

```c++
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

const int N = 1010;
int g[N][N],a[N][N];

void solve(){
  int n,m,k;
  cin>>n>>m>>k;
  for(int i=1;i<=n;i++){
    for(int j=1;j<=m;j++)
      scanf("%d",&g[i][j]);
  }
  for(int i=1;i<=n;i++){
    for(int j=1;j<=m;j++){
      a[i][j]=a[i-1][j]+a[i][j-1]+g[i][j]-a[i-1][j-1];
      // cout<<a[i][j]<<" ";
    }
    // puts("");
  }
  
  while(k--){
    int x1,y1,x2,y2;
    scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
    printf("%d\n",a[x2][y2]-a[x2][y1-1]-a[x1-1][y2]+a[x1-1][y1-1]);
  
  }
  
}
int main(){
  solve();
  return 0;
}
```

## 差分

* 一维差分

```c++
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

const int N = 1e5+10;

int a[N],b[N];

int main(){
  int n,k;
  cin>>n>>k;
  scanf("%d",&a[1]);
  for(int i=2;i<=n;i++){
    scanf("%d",&a[i]);
  }
  
  b[1]=a[1];
  for(int i=2;i<=n;i++) b[i]=a[i]-a[i-1];
  
  while(k--){
    int l,r,d;
    scanf("%d%d%d",&l,&r,&d);
    b[l]+=d,b[r+1]-=d;
  }
  
  
  for(int i=1;i<=n;i++)
    b[i]+=b[i-1];
  for(int i=1;i<=n;i++)
    printf("%d ",b[i]);
  puts("");
  return 0;
}
```


* 二维差分

```c++

#include <iostream>
#include <algorithm>

using namespace std;

const int N = 1e3+10;

int a[N][N],b[N][N];

void add(int x1,int y1,int x2,int y2,int d){
  //建议画图比较好理解
  b[x1][y1]+=d;
  b[x1][y2+1]-=d;
  b[x2+1][y1]-=d;
  b[x2+1][y2+1]+=d;
}

int main(){
  int n,m,q;
  cin>>n>>m>>q;
  for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
      scanf("%d",&a[i][j]);
  
  for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
        add(i,j,i,j,a[i][j]);
  
   

  while(q--){
    int x1,x2,y1,y2,c;
    scanf("%d%d%d%d%d",&x1,&y1,&x2,&y2,&c);
    add(x1,y1,x2,y2,c);
  
  }
  //前缀和
  for (int i = 1; i <= n; i ++ )
    for (int j = 1; j <= m; j ++ )
        b[i][j] += b[i - 1][j] + b[i][j - 1] - b[i - 1][j - 1];
        
  for(int i=1;i<=n;i++){
    for(int j=1;j<=m;j++)
      printf("%d ",b[i][j]);
    puts("");
  }
  
  return 0;
  
}
```
