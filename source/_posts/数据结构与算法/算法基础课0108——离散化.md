# 离散化


思想：


## 题目一


```c++
#include <iostream>
#include <algorithm>
#include <cstring>
#include <vector>

using namespace std;

const int N = 1e5+10;
typedef pair<int,int> pii;
vector<int> e;

int find(int x){
  int l=0,r=e.size();
  while(l<r){
    int mid = l+r>>1;
    if(e[mid]>x)r=mid;
    else l = mid+1;
  }
  return l;
}

int main(){
  
  int n,m;
  cin>>n>>m;
  
  vector<pii> add;
  vector<pii> query;
  
  for(int i=0;i<n;i++)
  {
    int x,y;
    cin>>x>>y;
    add.push_back({x,y});
    e.push_back(x);
  }
  for(int i=0;i<m;i++){
    int x,y;
    cin>>x>>y;
    query.push_back({x,y});
    e.push_back(x);
    e.push_back(y);
  }
  
  //去重
  sort(e.begin(),e.end());
  e.erase(unique(e.begin(),e.end()),e.end());
  
  int size = e.size();
  vector<int> axios(size+1);
  //把询问区间离散化
  for(int i=0;i<query.size();i++){
    int x = query[i].first,y=query[i].second;
    query[i]={find(x),find(y)};
  }
  //给数轴上数
  for(int i=0;i<add.size();i++){
    int x = add[i].first;
  
    axios[find(x)]+=add[i].second;
  }
  vector<int> s(size+1);
  for(int i=1;i<=size;i++){
    // cout<<axios[i-1]<<endl;
    s[i]=s[i-1]+axios[i];
    // cout<<s[i]<<endl;
  }
  
  for(auto q:query){
    // cout<<q.second<<" "<<q.first<<endl;
    printf("%d\n",s[q.second]-s[q.first-1]);
  }
  
  return 0;
}
```
