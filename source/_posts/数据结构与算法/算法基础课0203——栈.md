# 栈

**一种先入先出的数据结构**

## 题目1 模拟栈

```c++
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;
const int N = 1e5+10;
int stk[N],tt=-1;

void push(int x){
  stk[++tt]=x;
}

void pop(){
  if(tt>=0) tt--;
}

bool empty(){
  return tt==0;
}

int main(){
  
  int n;
  cin>>n;
  while(n--){
    char opr[6];
    scanf("%s",opr);
    int x;
  
    if(!strcmp(opr,"push")){
      scanf("%d",&x);
      push(x);
    }else if(!strcmp(opr,"pop")){
      pop();
    }
    else if(!strcmp(opr,"query")){
      printf("%d\n",stk[tt]);
    }else{
      if(tt==-1) puts("YES");
      else puts("NO");
    }
  }
  
  
  return 0;
}
```

## 题目二 中缀表达式求值

```c++
#include <iostream>
#include <algorithm>
#include <cstring>
#include <unordered_map>
#include <stack>

using namespace std;

stack<int> num;
//单调符号栈 栈中的符号优先级呈单调不增性
stack<char> op;

//将运算独立出来
void eval(){
  int x;
  int b = num.top();num.pop();
  int a = num.top();num.pop();
  char c = op.top();op.pop();
  
  if(c=='+')x = a+b;
  else if(c=='-') x=a-b;
  else if(c=='*') x=a*b;
  else x = a/b;
  num.push(x);
}

int main(){
  //定义算数运算符的优先级
  unordered_map<char,int> m={{'+',1},{'-',1},{'*',2},{'/',2}};
  
  string str;
  cin>>str;
  
  for(int i=0;i<str.size();i++){
    auto c = str[i];
    if(isdigit(c)){
      int x = 0;
      int j = i;
      // j记得++ 不然TLE
      while(j<str.size()&&isdigit(str[j])) x=x*10+str[j++]-'0';
      i = j-1;
      num.push(x);
    }else if(c=='('){
      op.push(c);
    }else if(c==')'){
      while(op.top()!='(') eval();
      op.pop();
    }else{
      // 若 栈中的算数运算符的优先级较高 则 需要先进行运算，不然会按照同级的处理。
      while(op.size()&&op.top()!='('&&m[op.top()]>=m[c]) eval();
      op.push(c);
    }
  }
  while(op.size()) eval();
  printf("%d",num.top());
  return 0;
}
```
