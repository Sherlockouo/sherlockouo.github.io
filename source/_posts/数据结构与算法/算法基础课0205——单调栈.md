## 题目一 模版

```c++
#include <iostream>
#include <deque>

using namespace std;

int main(){
  deque<int> q;
  int n;
  cin>>n;
  while(n--){
    int x;
    cin>>x;
    while(q.size()&&q.back()>=x) q.pop_back();
    if(!q.size()) printf("-1 ");
    else{
      if(q.back()==x) printf("-1 ");
      else printf("%d ",q.back());
    }
    q.push_back(x);
  }
  
  return 0;
}
```
