## 第一题 滑动窗口

**题意：** 求窗口中的最小值，最大值，在每次滑动窗口的时候，输出这个值


```c++
#include <iostream>
#include <algorithm>
#include <cstring>
#include <deque>

using namespace std;
const int N =1e6+10;
int a[N];
int main(){
  int n,k;
  cin>>n>>k;
  for(int i=0;i<n;i++)
    scanf("%d",&a[i]);
  deque<int> mx,mn;
  for(int i=0;i<n;i++){
    while(mn.size()&&mn.front()+k<=i) mn.pop_front();
    while(mn.size()&&a[mn.back()]>=a[i]){
      mn.pop_back();
    }
    mn.push_back(i);
    if(i>=k-1)
      printf("%d ",a[mn.front()]);
  }
  puts("");
  for(int i=0;i<n;i++){
    while(mx.size()&&mx.front()+k<=i) mx.pop_front();
    while(mx.size()&&a[mx.back()]<=a[i]){
      mx.pop_back();
    }
    mx.push_back(i);
    if(i>=k-1)
      printf("%d ",a[mx.front()]);   
  }
  puts("");
  return 0;
}
```
