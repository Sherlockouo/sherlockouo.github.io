## 简单模版(字符串版yxc)

```c++
#include <iostream>

using namespace std;

const int N = 1e5+10;

char s[N];
int son[N][26],cnt[N],idx;
int n;

void insert(char *str){
  int par = 0;
  for(int i=0;str[i];i++){
    int u = str[i]-'a';
    if(!son[par][u]) son[par][u]=++idx;
    par = son[par][u];
  }
  cnt[par]++;
}

int query(char *str){
  int p = 0;
  for(int i=0;str[i];i++){
    int u = str[i]-'a';
    if(!son[p][u]) return 0;
    p = son[p][u];
  }
  return cnt[p];
}

int main(){
  cin>>n;
  while(n--){
    char opr[1];
    scanf("%s%s",opr,s);
    if(*opr=='I'){
      insert(s);
    }else{
      cout<<query(s)<<endl;
    }
  }
  return 0;
}

```

## 题目1 最大异或对(01字典树)

```c++
#include <iostream>

using namespace std;

const int N = 1e5+10,M = 3100010;
int n;
int son[M][2],a[N],idx;

void insert(int x){
  int p = 0;
  for(int i=30;i>=0;i--){
    int &s = son[p][x>>i&1];
    if(!s) s = ++idx;
    p = s;
  }
}

int query(int x){
  int res =0;
  int p = 0;
  for(int i=30;i>=0;i--){
  
    int s = x>>i&1;
  
    if(son[p][!s]){
      res+=1<<i;
      p=son[p][!s];
    }else p = son[p][s];
  }
  return res;
}

int main(){
  
  cin>>n;
  for(int i=0;i<n;i++){
    scanf("%d",&a[i]);
    insert(a[i]);
  }
  
  int ans  = 0;
  for(int i=0;i<n;i++) ans = max(ans,query(a[i]));
  
  printf("%d",ans);
  
  return 0;
}
```
