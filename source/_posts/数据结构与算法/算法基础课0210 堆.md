## 题目1 堆排序

**下沉建堆** 时间复杂度 o(logn)

只需要一个down函数即可 ，贪心的比较 交换即可。

```c++
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 1e5+10;

//堆 a 堆的大小 sz
int a[N],sz;

void down(int u){
  int t = u;
  // 若 当前节点的左儿子存在 且 左儿子小于当前节点 则 将 左儿子的节点赋值给t 作为交换 的节点
  if(u*2<=sz&&a[u*2]<a[t]) t = u*2;
  // 同理处理右节点
  if(u*2+1<=sz&&a[u*2+1]<a[t]) t= u*2+1;
  if(u!=t){
    swap(a[u],a[t]);
    down(t);
  }
}


int main(){
  int n,k;
  cin>>n>>k;
  sz=n;
  for(int i=1;i<=n;i++) scanf("%d",&a[i]);
  
  for(int i=n/2;i;i--)  down(i);
  
  while(k--){
    printf("%d ",a[1]);
    a[1]=a[sz];
    sz--;
    down(1);
  }
  
  return 0;
}
```


## 题目二 模拟堆

**实现操作**


```c++
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

const int N = 1e5+10;

int n,ph[N],hp[N],h[N],sz,m;

void heapSwap(int a,int b){
  swap(ph[hp[a]],ph[hp[b]]);
  swap(hp[a],hp[b]);
  swap(h[a],h[b]);
}

void down(int u){
  int t = u;
  //这里需要跟t比较大小 若 根>左>右 如果用2*u+1 和 u比较的话就不能达到目的。
  if(u*2<=sz&&h[u*2]<h[t]) t = u*2;
  if(u*2+1<=sz&&h[u*2+1]<h[t]) t = u*2+1;
  if(u!=t){
    heapSwap(u,t);
    down(t);
  }
}

void up(int u){
  while(u/2>0&&h[u/2]>h[u]){
    heapSwap(u,u/2);
    u>>=1;
  }
}

int main(){
  cin>>n;
  while(n--){
    char opr[5];
    scanf("%s",opr);
    int x,k;
    if(!strcmp(opr,"I")){
      cin>>x;
      sz++;
      m++;
      //ph 从第几个数 知道在堆中的下标
      //hp 从堆中的下标 直到是第几个插入的数
      //m -> sz sz->m
      ph[m]=sz,hp[sz]=m;
  
      h[sz]=x;
      up(sz);
    }else if(!strcmp(opr,"PM")){
      printf("%d\n",h[1]);
    }else if(!strcmp(opr,"DM")){
      heapSwap(1,sz);
      sz--;
      down(1);
    }else if(!strcmp(opr,"D")){
      cin>>k;
      k = ph[k];
      heapSwap(k,sz);
      sz--;
      down(k),up(k);
    }else{
      cin>>k>>x;
      k = ph[k];
      h[k]=x;
      up(k);
      down(k);
    }
  }
  return 0;
}
```
