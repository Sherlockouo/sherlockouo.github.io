**开放寻址法**


```c++
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

const int N = 200010,null=0x3f3f3f3f;
int h[N];

int find(int x){
  int t = (x%N+N)%N;
  while(h[t]!=null&&h[t]!=x){
    t++;
    if(t==N) t =0;
  }
  return t;
}

int main(){
  memset(h,0x3f,sizeof h);
  int n;
  cin>>n;
  while(n--){
    char c[2];
    int k;
    scanf("%s%d",c,&k);
    if(!strcmp(c,"I")){
      h[find(k)]=k;
    }else{
      if(h[find(k)]==null){
        printf("No\n");
      }else{
        printf("Yes\n");
      }
    }
  }
  return 0;
}
```
