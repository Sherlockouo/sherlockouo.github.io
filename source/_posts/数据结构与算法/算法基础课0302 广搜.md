# 广度优先搜索

**一般用于求最短路或者最优解**

## 题目1 八数码


> 1. 用字符串表示每一个状态
> 2. 最短距离就是每次移动的最少次数
> 3. 状态转移：'x'向四个方向移动，若移动的结果在map中没出现过，那么加入队列(即一个新状态)

```
#include <iostream>
#include <algorithm>
#include <queue>
#include <unordered_map>

using namespace std;


unordered_map<string,int> dict;

int bfs(string start){
  string end = "12345678x";
  queue<string> q;
  q.push(start);
  dict[start]=0;
  
  int dx[4]={-1,0,1,0};
  int dy[4]={0,1,0,-1};
  
  while(q.size()){
   auto t = q.front();
   q.pop();
   int distance = dict[t];
  // cout<<t<<" "<<end<<endl;
   if(t==end) return dict[t];
   
   int p = t.find('x');
   int x = p/3,y= p%3;
   
   for(int i=0;i<4;i++){
     int a=x+dx[i],b=y+dy[i];
     if(a>=3||b>=3||a<0||b<0) continue;
     swap(t[p],t[3*a+b]);
  
     if(!dict.count(t)){
       dict[t]=distance+1;
       q.push(t);
     }
     swap(t[p],t[3*a+b]);
   
   }
  
  }
  return -1;
}

int main(){
  string start;
  for(int i=0;i<9;i++){
    char c;
    cin>>c;
    start+=c;
  }
  
  cout<<bfs(start)<<endl;
  return 0;
}
```
