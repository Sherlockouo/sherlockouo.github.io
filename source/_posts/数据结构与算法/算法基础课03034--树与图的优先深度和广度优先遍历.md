# 深广遍历


## 题目1 树的重心


```
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;
const int N = 1e5+10,M=2*N;
int e[M],h[N],ne[M],idx;
int vis[N],ans=N;
int n;

void add(int a, int b)  // 添加一条边a->b
{
    e[idx] = b, ne[idx] = h[a], h[a] = idx ++ ;
}

int dfs(int p){
  vis[p]=1;
  int sum = 0,sz = 0;
  for(int i = h[p];i!=-1;i=ne[i]){
    int j = e[i];
    if(vis[j]) continue;
    int s = dfs(j);
    sz = max(sz,s);
    sum+=s;
  }
  sz = max(sz,n-sum-1);
  ans = min(ans,sz);
  return sum+1;
}

int main(){
  cin>>n;
  memset(h,-1,sizeof h);
  for(int i=0;i<n-1;i++){
    int a,b;
    scanf("%d%d", &a, &b);
    add(a, b);
    add(b,a);
  }
  
  dfs(1);
  printf("%d\n",ans);
  return 0;
}

```

## 题目2 图中点的层次

bfs 模版

```
#include <iostream>
#include <cstring>
#include <algorithm>
#include <queue>

using namespace std;

const int N = 1e5+10,M=2*N;
int h[N], e[M], ne[M], idx;
int n,m,vis[N];

void add(int a, int b)  // 添加一条边a->b
{
    e[idx] = b, ne[idx] = h[a], h[a] = idx ++ ;
}

int bfs(int u){
  queue<int> q;
  q.push(u);
  int minsteps = 0;
  while(q.size()){
    int sz = q.size();
    while(sz--){
      int t = q.front();
      if(t==n) return minsteps;
      q.pop();
      vis[t] = 1;
  
      for(int i = h[t];i!=-1;i=ne[i]){
        int j = e[i];
        if(vis[j]) continue;
        q.push(j);
      }
    }
    minsteps++;
  }
  return -1;
}

int main(){
  cin>>n>>m;
  memset(h,-1,sizeof h);
  for(int i=0;i<m;i++){
    int a,b;
    scanf("%d%d",&a,&b);
    add(a, b);
  }
  
  cout<<bfs(1)<<endl;
  
  return 0;
}
```
