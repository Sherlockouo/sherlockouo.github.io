# 拓扑排序

**解决依赖问题**


## 题目1 有向图的拓扑排序

- 先把入度为零的点放入队列
- 遍历队列，不断减小点的入读
- 若所有点的数量cnt<n-1 则说明又环 无法拓扑排序

```
#include <iostream>
#include <algorithm>
#include <cstring>
#include <vector>
#include <queue>

using namespace std;
const int N = 1e5+10,M=2*N;
int e[M],h[N],ne[M],idx;
int indeg[N];
int n,m;

void add(int a, int b)  // 添加一条边a->b
{
    e[idx] = b, ne[idx] = h[a], h[a] = idx ++ ;
}


int main(){
  cin>>n>>m;
  memset(h,-1,sizeof h);
  while(m--){
    int a,b;
    scanf("%d%d",&a,&b);
    add(a,b);
    indeg[b]++;
  }
  queue<int> q;
  vector<int> v;
  for(int i=1;i<=n;i++) if(!indeg[i]) q.push(i);
  int cnt = 0;
  while(q.size()){
    int sz = q.size();
    while(sz--){
      auto t = q.front();
      q.pop();
      v.push_back(t);
      for(int i = h[t];i!=-1;i=ne[i]){
        int j = e[i];
        if(!(--indeg[j])){
          q.push(j);
           cnt++;
        }
      }
    }
  }
  // printf("%d %d\n",cnt,n);
  if(cnt!=n-1) printf("-1\n");
  else {
    for(auto x:v){
      printf("%d ",x);
    }
    puts("");
  }
  return 0;
}

```
