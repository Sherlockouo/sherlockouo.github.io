## 1.Djkstra

- **用于有向图和无向图，所有边权都为正的图，求单源最短路**

* 最短路满足最优子结构性质，最优子结构可以描述为：D(s,t)=D(s->v1->v2->t)

```
//to-be practised
#include <iostream>
#include <algorithm>
#include <cstring>
#include <queue>
#define x first
#define y second

using namespace std;

typedef pair<int, int> PII;
const int N = 510,M=1e5+10;
int h[N], e[M], w[M], ne[M], idx;
int dist[N];
bool st[N];
int n,m;
int g[N][N];
void add(int a,int b,int c){
  e[idx]=b,w[idx]=c,ne[idx]=h[a],h[a]=idx++;
}

int dj(){
  //初始化为+∞
  memset(dist,0x3f,sizeof dist);
  //重置访问记录
  memset(st,0,sizeof st);
  //从第一个节点开始遍历
  dist[1]=0;
  //遍历剩下的n-1个点
  for(int i=0;i<n-1;i++){
    int t = -1;
    //从中选出最短的一条边
    for(int j=1;j<=n;j++){
      if(!st[j]&&(t==-1||dist[t]>dist[j])) t = j;
    }
    //计算当前节点到所有节点的最小值
    //更新所有i能达到的节点 计算从源点到j点的最短路径
   //下次选的也就是到所有点的最短路 若没有边则为+∞ 不会被选为最短路
    for(int j=1;j<=n;j++)
      dist[j]=min(dist[j],dist[t]+g[t][j]);
      st[t]=1;
  }
  
  if(dist[n]==0x3f3f3f3f) return -1;
  return dist[n];
}

int main(){
  cin>>n>>m;
  memset(g,0x3f,sizeof g);
  while(m--){
    int a,b,c;
    scanf("%d%d%d",&a,&b,&c);
    // add(a, b, c);
  
    g[a][b]=min(g[a][b],c);
    // g[b][a]=c;
    // add(b,a,c);
  }
  cout<<dj()<<endl;
  
  return 0;
}
```
