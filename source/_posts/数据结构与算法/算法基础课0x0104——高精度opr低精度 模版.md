  ```
        s += s1[j]-'0';
        ans+=s%10+'0';
        if(s!=0)
          s/=10;
      }
    if(s!=0)
    ans+=s+'0';
    reverse(ans.begin(),ans.end());
    return ans;

  }
  ```

## 减法

这个算法求的是对于 两个正数的减法,负数的话,可以转化为正数来算.

为了少些边界条件判断,所以这么写

```c++
#include <iostream>
#include <algorithm>
#include <cstring>
#include <vector>

using namespace std;
vector<int> c;
bool cmp(vector<int>& a,vector<int> &b){
  if(a.size()!=b.size()) return a.size()>b.size();
  for(int i=a.size()-1;i>=0;i--)
    if(a[i]!=b[i])
      return a[i]>b[i];
  
  return true;
}
void prt(vector<int> &a){
  for(auto x:a){
    cout<<x;
  }
  cout<<endl;
}
void solve(vector<int> &a,vector<int> &b){
  for(int i=0,t=0;i<a.size();i++){
    t=a[i]-t;
    if(i<b.size())t -= b[i];
    c.push_back((t+10)%10);
    if(t<0) t = 1;
    else t = 0;
  }
  while(c.size()>1 && c.back()==0){
    c.pop_back();
  } 
}

int main(){
  string s1,s2;
  cin>>s1>>s2;
  int l1 = s1.size(),l2=s2.size();
  vector<int> a,b;
  for(int i=s1.size()-1;i>=0;i--) a.push_back(s1[i]-'0');
  for(int i=s2.size()-1;i>=0;i--) b.push_back(s2[i]-'0');
  
  if(cmp(a,b)) solve(a,b);
  else{
    solve(b,a);
    printf("-");
  }
  for(int i=c.size()-1;i>=0;i--){
    printf("%d",c[i]);
  }
  
  return 0;
}
```

```
vector<int> mul(vector<int> &a,int x){
  // prt(a);
  vector<int> c;
  //进位数
  int t =0;
  //从低位到高位相乘
  for(int i=0;i<a.size()||t;i++){
    if(i<a.size()) t+=a[i]*x;
    c.push_back(t%10);
    t/=10;
  }
  while(c.size()>1&&c.back()==0) c.pop_back();
  
  return c;
}

int main(){
  string s;
  int x;
  cin>>s>>x;
  
  vector<int> a;
  for(int i=s.size()-1;i>=0;i--) a.push_back(s[i]-'0');
  
  auto c = mul(a,x);
  for(int i=c.size()-1;i>=0;i--) printf("%d",c[i]);

  return 0;
```
